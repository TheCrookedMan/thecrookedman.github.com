JavaScript 类型转换

  • JavaScript

ECMAScript 有 5 种原始类型，Undefined、Null、Boolean、Number 和 String。 原始类型详解

其他类型转化为数字

Undefined    ==>     Number

var a; //undefined
var num = new Number(a);
console.log(num.valueOf()); //返回 NaN


var a; //undefined
console.log(+a); //返回 NaN

Null    ==>     Number

var a = null; //null
var num = new Number(a);
console.log(num.valueOf()); //返回 0


var a = null; //null
console.log(+a); //返回 0

Boolean    ==>     Number

var boo = false;
console.log(+boo); // 0

var foo = true;
console.log(+foo); // 1

String    ==>     Number

var n1 = "123456";
console.log(+n1); // 123456

var n2 = "zzzzzzzz";
console.log(+n2); // NaN

不同类型的基础数据之间的加法，数据先转换为number，然后转换为string(如果有string类型数据参与运算)

+运算符

如果某个运算数是字符串，那么采用下列规则：
如果两个运算数都是字符串，把第二个字符串连接到第一个上。
如果只有一个运算数是字符串，把另一个运算数转换成字符串，结果是两个字符串连接成的字符串。
原理：首先调用对象中的 valueOf() 方法转换为符合相关运算的值。如果转换的值无效在调用对象中的 toString() 方法转换为字符窜。

var al = null; //null
var bl; //undefined
console.log(al+bl); // NaN


var cl = false;
console.log(al+cl); // 0+0 = 0

var dl = 1;
var el = true;
console.log(dl+el); // 1 + 1 = 2

var fl = "1";
console.log(dl+fl); // 1 + "1" = "11"


var gl = 3;
var vl = 7;
var ll = "0";
console.log(gl+vl+ll); // 3 + 7 + "0" = 10 + "0" = "100"

var gla = 3;
var vla = 7;
var lla = "0";
console.log( gla + ( vla+lla ) ); // 3 + (7 + "0") = 3 + "70" = "370"

var str = "str";
var sn = null;
console.log(str + sn); // "strnull"

var str = "str";
var sn;
console.log(str + sn); // "strundefined"

var str = "str";
var sn = true;
console.log(str + sn); // "strtrue"

var str = "str";
var sn = true;
console.log(sn + str); // "truestr"

var str = "str";
var sn = true;
var nn = 0;
console.log(sn + nn + str); // 0 + true + "str" = 0 + 1 + "str" = "1" + "str" = "1str "

var nm = 4;
var bn = true;
var sh;
var str = "scc";
console.log(nm + bn + sh + str); // 4 + true + undefined + "scc" = 4 + 1 + undefined + "scc" = 5 + undefined + "scc" = NaN + "scc" = "NaNscc"

对象参与加减法运算

对象参与基础类型数据运算，先转化为基础类型。先调用其valueOf方法，如果返回的不是基础类型，再调用其toString方法,如果返回的还不是基础类型，则抛出错误。但是，Date数据刚好相反

Date.prototype.toString = function(){
    console.log("这里是 toString 方法。");
    return 2;
}

Date.prototype.valueOf = function(){
    console.log("这里是 valueOf 方法。");
    return 3;
}

var a = new Date(),b = new Date();

console.log( a + b); //调用toString，2 + 2，结果是4

// 再重写toString方法

Date.prototype.toString = function(){
    console.log("这里是 toString 方法。");
    return {};
}

var c = new Date,d = new Date;

console.log( c + d ); // 调用toString方法返回的不是基础类型，再调用valueOf，3 + 3，结果是6

// 再重写valueOf方法
Date.prototype.valueOf = function(){
    console.log("这里是 valueOf 方法。");
    return {};
}

var e = new Date,f = new Date;

console.log( e + f ); // TypeError: Cannot convert object to primitive value

数据前有加号‘+’，可以让字符串转化为数字

console.log(+"10000"+86); // 10086

console.log(+"str"+86); // NaN

Js中提供 parseInt() 方法 将数值转换为 整数

parseInt("123456");  // 123456
parseInt(0);  // 0
parseInt("0");  // 0
parseInt("123.456");  // 123
parseInt("-123.456");  // -123
parseInt('06'); // 6

parseInt('08'); // 8 ，但是早期一些浏览器返回的是 0 。大部分浏览器都已修正这个问题。如果低版本的浏览器出现这个问题使用  parseInt("08",10) .

parseInt("500g"); // 500
parseInt("g500"); // NaN

我们可以利用 ECMAScript 运算符的特性来自己写一个parseInt()方法。

function parseInt(n) {
    return n*1^0^0 || 0;
}

//验证
parseInt("123456");  // 123456
parseInt(0);  // 0
parseInt("0");  // 0
parseInt("123.456");  // 123
parseInt("-123.456");  // -123
parseInt('06'); // 6

parseInt('08'); // 8

parseInt("500g"); // 0
parseInt("g500"); // 0

注意了，以上函数能表示的整数有效范围是 -2147483648 ~ 2147483648
为了在js中表达更大范围的数值，还有一个函数提供参考，如下：

function parseInt(number) {
    return Infinity === number ? 0 : (number*1 || 0).toFixed(0)*1;
}

其他类型转换为布尔值

console.log(!!5);           // true
console.log(!!NaN);         // false
console.log(!!undefined);   // false
console.log(!!null);        // false
console.log(!!false);       // false
console.log(!!-100);        // true
console.log(!!function(){});// true

JS 中三个取整函数的区别：

// Math.ceil();取大于等于参数的最小整数。

console.log(Math.ceil(8.7));  // 9
console.log(Math.ceil(-8.7)); // -8

// Math.floor():取小于等于参数的最小整数。

console.log(Math.floor(8.7));  // 8
console.log(Math.floor(-8.7)); // -9

//Math.round():"四舍五入"取整数。

console.log(Math.round(8.7));  // 9
console.log(Math.round(-8.7)); // -9
